        // Smooth Scroll for Navigation Links
        document.querySelectorAll('a[href^="#"]').forEach(anchor => {
            anchor.addEventListener('click', function (e) {
                e.preventDefault();
                document.querySelector(this.getAttribute('href')).scrollIntoView({
                    behavior: 'smooth'
                });
                console.log(`Smooth scrolling to ${this.getAttribute('href')}`);
            });
        });

        // Basic Form Validation with Debugging
        document.querySelector('form').addEventListener('submit', function (e) {
            const name = document.querySelector('input[name="name"]');
            const email = document.querySelector('input[name="email"]');
            const phone = document.querySelector('input[name="phone"]');
            const message = document.querySelector('textarea[name="message"]');

            let valid = true;

            // Reset all fields' border color
            [name, email, phone, message].forEach(field => {
                field.style.borderColor = '';
            });

            if (!name.value) {
                name.style.borderColor = 'red';
                valid = false;
                console.log('Name field is empty');
            }

            if (!email.value || !/\S+@\S+\.\S+/.test(email.value)) {
                email.style.borderColor = 'red';
                valid = false;
                console.log('Invalid or empty email');
            }

            if (!phone.value) {
                phone.style.borderColor = 'red';
                valid = false;
                console.log('Phone field is empty');
            }

            if (!message.value) {
                message.style.borderColor = 'red';
                valid = false;
                console.log('Message field is empty');
            }

            if (!valid) {
                e.preventDefault();
                alert('Please fill in all fields correctly');
                console.log('Form submission prevented due to invalid inputs');
            } else {
                console.log('Form submitted successfully');
            }
        });
    </script>
